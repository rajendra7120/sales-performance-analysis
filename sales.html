import pandas as pd import numpy as np import matplotlib.pyplot as plt import seaborn as sns from sklearn.preprocessing import StandardScaler from statsmodels.tsa.arima.model import ARIMA # Load and merge data sales_df = pd.read_csv('sales_transactions.csv') customers_df = pd.read_csv('customers.csv') products_df = pd.read_csv('products.csv') # Merge datasets df = sales_df.merge(customers_df, on='customer_id', how='left') df = df.merge(products_df, on='product_id', how='left') # Data cleaning df['order_date'] = pd.to_datetime(df['order_date']) df['revenue'] = df['quantity'] * df['price'] df['profit'] = df['revenue'] - (df['quantity'] * df['cost']) df.dropna(subset=['customer_id', 'product_id'], inplace=True) # Calculate YoY growth df['year'] = df['order_date'].dt.year df['month'] = df['order_date'].dt.month monthly_sales = df.groupby(['year', 'month'])['revenue'].sum().reset_index() monthly_sales['yoy_growth'] = monthly_sales.groupby('month')['revenue'].pct_change() * 100 # RFM Analysis snapshot_date = df['order_date'].max() + pd.Timedelta(days=1) rfm = df.groupby('customer_id').agg({ 'order_date': lambda x: (snapshot_date - x.max()).days, # Recency 'order_id': 'count', # Frequency 'revenue': 'sum' # Monetary }).rename(columns={ 'order_date': 'Recency', 'order_id': 'Frequency', 'revenue': 'Monetary' }) # Score RFM segments rfm['R_Score'] = pd.qcut(rfm['Recency'], 4, labels=[4,3,2,1]) rfm['F_Score'] = pd.qcut(rfm['Frequency'].rank(method='first'), 4, labels=[1,2,3,4]) rfm['M_Score'] = pd.qcut(rfm['Monetary'], 4, labels=[1,2,3,4]) rfm['RFM_Score'] = rfm[['R_Score','F_Score','M_Score']].sum(axis=1) # Time series forecasting ts_data = df.groupby('order_date')['revenue'].sum() model = ARIMA(ts_data, order=(2,1,2)) results = model.fit() forecast = results.forecast(steps=90) # 90-day forecast print(f"Total Revenue: ${df['revenue'].sum():,.2f}") print(f"Average Order Value: ${df['revenue'].mean():,.2f}") print(f"Customer Count: {df['customer_id'].nunique():,}")

-- Year-over-Year Growth Analysis WITH monthly_sales AS ( SELECT DATE_FORMAT(order_date, '%Y-%m') AS month, YEAR(order_date) AS year, SUM(quantity * price) AS total_revenue, COUNT(DISTINCT order_id) AS order_count FROM sales_transactions GROUP BY month, year ), yoy_comparison AS ( SELECT m1.month, m1.year AS current_year, m1.total_revenue AS current_revenue, m2.total_revenue AS previous_revenue, ((m1.total_revenue - m2.total_revenue) / m2.total_revenue * 100) AS yoy_growth_pct FROM monthly_sales m1 LEFT JOIN monthly_sales m2 ON MONTH(m1.month) = MONTH(m2.month) AND m1.year = m2.year + 1 ) SELECT * FROM yoy_comparison ORDER BY current_year DESC, month DESC; -- Top Customers by Revenue (RFM High Value) SELECT c.customer_id, c.customer_name, c.region, COUNT(DISTINCT s.order_id) AS total_orders, SUM(s.quantity * s.price) AS total_revenue, AVG(s.quantity * s.price) AS avg_order_value, DATEDIFF(CURDATE(), MAX(s.order_date)) AS days_since_last_order FROM customers c JOIN sales_transactions s ON c.customer_id = s.customer_id GROUP BY c.customer_id, c.customer_name, c.region HAVING total_revenue > 10000 ORDER BY total_revenue DESC LIMIT 100; -- Regional Performance Dashboard SELECT r.region_name, COUNT(DISTINCT s.order_id) AS total_orders, COUNT(DISTINCT s.customer_id) AS unique_customers, SUM(s.quantity * s.price) AS total_revenue, SUM(s.quantity * (s.price - p.cost)) AS total_profit, (SUM(s.quantity * (s.price - p.cost)) / SUM(s.quantity * s.price) * 100) AS profit_margin_pct, r.sales_target, (SUM(s.quantity * s.price) / r.sales_target * 100) AS target_achievement_pct FROM regions r LEFT JOIN customers c ON r.region_id = c.region_id LEFT JOIN sales_transactions s ON c.customer_id = s.customer_id LEFT JOIN products p ON s.product_id = p.product_id WHERE YEAR(s.order_date) = 2024 GROUP BY r.region_name, r.sales_target ORDER BY total_revenue DESC; -- Product ABC Analysis WITH product_revenue AS ( SELECT p.product_id, p.product_name, p.category, SUM(s.quantity * s.price) AS total_revenue, SUM(s.quantity) AS total_quantity FROM products p JOIN sales_transactions s ON p.product_id = s.product_id WHERE YEAR(s.order_date) = 2024 GROUP BY p.product_id, p.product_name, p.category ), ranked_products AS ( SELECT *, SUM(total_revenue) OVER () AS grand_total, SUM(total_revenue) OVER (ORDER BY total_revenue DESC) AS cumulative_revenue, (SUM(total_revenue) OVER (ORDER BY total_revenue DESC) / SUM(total_revenue) OVER ()) * 100 AS cumulative_pct FROM product_revenue ) SELECT *, CASE WHEN cumulative_pct <= 70 THEN 'A' WHEN cumulative_pct <= 90 THEN 'B' ELSE 'C' END AS abc_category FROM ranked_products ORDER BY total_revenue DESC;
